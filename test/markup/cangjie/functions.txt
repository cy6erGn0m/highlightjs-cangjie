func simple(): Int64 {
    return 42
}

func withParams(a: Int64, b: Int64): Int64 {
    return a + b
}

func namedParams(name!: String, age!: Int64): String {
    return "${name}: ${age}"
}

func defaultValue(a: Int64 = 0): Int64 {
    return a
}

func generic<T>(value: T): T {
    return value
}

func multiGeneric<T, U>(first: T, second: U): (T, U) {
    return (first, second)
}

public func publicFunc(): Unit {}
private func privateFunc(): Unit {}
protected func protectedFunc(): Unit {}
internal func internalFunc(): Unit {}

mut func mutFunc(): Unit {}
static func staticFunc(): Unit {}
override func overrideFunc(): Unit {}
redef func redefFunc(): Unit {}
abstract func abstractFunc(): Unit

operator func +(a: Int64, b: Int64): Int64 {
    return a + b
}

func lambda(): Unit {
    let add = (a: Int64, b: Int64): Int64 => a + b
    let square = (x: Int64): Int64 => x * x
    let noParam = (): Int64 => 42
    let closure = (x: Int64): Int64 => {
        let y = x + 1
        return y
    }
}

func nested(): Unit {
    func helper(x: Int64): Int64 {
        return x * 2
    }
    return helper(21)
}

init(x: Int64, y: Int64) {
    this.x = x
    this.y = y
}

~init() {
    println("Destroyed")
}

const func constFunc(): Int64 {
    return 42
}

func varParam(var a: Int64): Int64 {
    a = a + 1
    return a
}

func letParam(let a: Int64): Int64 {
    return a
}

func inoutParam(a: inout Int64): Unit {
    a = a + 1
}
