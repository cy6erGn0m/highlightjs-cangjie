macro package example.macros

import std.ast.*

public macro dprint(input: Tokens): Tokens {
    let inputStr = input.toString()
    let result = quote(
        print($(inputStr) + " = ")
        println($(input))
    )
    return result
}

public macro WithModifier(attr: Tokens, input: Tokens): Tokens {
    return attr + input
}

macro simple(input: Tokens): Tokens {
    return input
}

macro func debugLog(message: String): Unit {
    println(message)
}

public macro func trace(funcName: String, args: Tokens): Tokens {
    return quote {
        println("Entering: " + $(funcName))
        $(args)
        println("Exiting: " + $(funcName))
    }
}
