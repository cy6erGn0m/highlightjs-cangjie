public class Box<T> where T <: Comparable<T> {
    private let value: T
    
    public init(value: T) {
        this.value = value
    }
    
    public func getValue(): T {
        return value
    }
}

struct Point {
    let x: Float64
    let y: Float64
    
    func distance(): Float64 {

    }

6. Pattern Matching

Match expressions

func describe(x: Int64): String {
    match (x) {
        case 0 => "zero"
        case 1 => "one"
        case n if n < 0 => "negative"
        case n if n > 100 => "large"
        case _ => "other"
    }
}

7. Macros

Macro definitions and quote expressions

macro package example_macros

macro func logging(func_name: String) {
    quote(
        func ${func_name}() {
            println("Entering ${func_name}")
            let result = ${func_name}_impl()
            println("Exiting ${func_name}")
            return result
        }
    )
}
